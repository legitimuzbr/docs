<ResponseField
  name="mount"
  type="function(): void"
>
  É responsável por montar o SDK e deve ser chamado após a configuração.

  Exemplo:

  ```html HTML
  <div id="legitimuz-root"></div>
  <script src="https://cdn.legitimuz.com/js/sdk/faceindex.js"></script>
  <script type="text/javascript">
    const sdkInstance = LegitimuzFaceIndex({
      /* ...opções */
    });

    // Montar SDK
    sdkInstance.mount();
  </script>
  ```
</ResponseField>

<ResponseField
  name="setLang"
  type="function(lang: 'pt' | 'en' | 'es'): void"
>
  É responsável por alterar o idioma do fluxo de [Face Index](/introduction).

  Exemplo:

  <CodeGroup>
    ```html HTML
    <div id="legitimuz-root"></div>
    <script src="https://cdn.legitimuz.com/js/sdk/faceindex.js"></script>
    <script type="text/javascript">
      const sdkInstance = LegitimuzFaceIndex({
        /* ...opções */
      });

      // Montar SDK
      sdkInstance.mount();

      // Alterar idioma
      sdkInstance.setLang('en');
    </script>
    ```
  </CodeGroup>
</ResponseField>

<ResponseField
  name="closeModal"
  type="function(): void"
>
  É responsável por fechar o modal de [Face Index](/introduction).

  Exemplo:

  <CodeGroup>
    ```html HTML
    <div id="legitimuz-root"></div>
    <script src="https://cdn.legitimuz.com/js/sdk/faceindex.js"></script>

      const sdkInstance = LegitimuzFaceIndex({
        /* ...opções */
      });

      // Montar SDK
      sdkInstance.mount();

      // Fechar modal de verficação
      sdkInstance.closeModal();
    </script>
    ```
  </CodeGroup>
</ResponseField>

<ResponseField
  name="changeFieldId"
  type="function(params: object): void"
>
  Alterar o [id](https://developer.mozilla.org/pt-BR/docs/Web/HTML/Global_attributes/id)
  do campo usado para extrair o CPF.

  <Expandable title="params: object">
    <ResponseField name="fieldName"  type="string" required>
      Nome do campo

      [Clique aqui](/fields) para ver os campos disponíveis
    </ResponseField>
    <ResponseField name="fieldId"  type="string" required>
      Novo id do campo

      Exemplo: `meu-id-customizado`
    </ResponseField>
  </Expandable>

  Exemplo:

  <CodeGroup>
    ```html HTML
    <div id="legitimuz-root"></div>
    <script src="https://cdn.legitimuz.com/js/sdk/faceindex.js"></script>
    <script type="text/javascript">
      const sdkInstance = LegitimuzFaceIndex({
        /* ...opções */
      });

      // Montar SDK
      sdkInstance.mount();

      // Alterar id do campo cpf
      sdkInstance.changeFieldId({
        fieldName: "cpf",
        fieldId: "meu-id-customizado",
      });
    </script>
    ```
  </CodeGroup>
</ResponseField>

<ResponseField
  name="startFaceIndex"
  type="function(params: object): void"
>
  Iniciar o fluxo de [Face Index](/introduction) via javascript.

  <Expandable title="params: object">
    <ResponseField name="cpf"  type="string" required>
      CPF do usuário

      Exemplo: `12345678901`

      [Clique aqui](/fields) para ver os campos disponíveis
    </ResponseField>
    <ResponseField name="referenceId" type="string | null" default='null'>
      Campo usado para `identificadoes` customizados

      Exemplo: `fake-user-id`
    </ResponseField>
  </Expandable>

  Exemplo:

  <CodeGroup>
    ```html HTML
    <div id="legitimuz-root"></div>
    <script src="https://cdn.legitimuz.com/js/sdk/faceindex.js"></script>
    <script type="text/javascript">
      const sdkInstance = LegitimuzFaceIndex({
        /* ...opções */
      });

      // Montar SDK
      sdkInstance.mount();

      sdkInstance.startFaceIndex({ cpf: "12345678901" });
    </script>
    ```
  </CodeGroup>
</ResponseField>
